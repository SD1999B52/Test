"git --version" - версия Git
"git config user.email "d.shalupin@mail.ru"" - задать адрес автора
"git config user.name "Danil"" - задать имя автора
"cd /d path" - переход в каталог
"git init" - создать репозиторий
"git status" - состояние рабочего каталога и раздела проиндексированных файлов
"git add file" - добавить файл в отслеживаемые
"git add ." - добавить все файлы в каталоге в отслеживаемое
"git commit" - сделать снимок проекта
"git log" - отобразить сделанные снимки проекта
"git rm file" - удалить файл из отслеживаемых
"git log --oneline" - отобразить сделанные снимки проекта в компактном виде
"git checkout hash" - переход к определенному снимку
"git diff --staged" - увидеть изменения, которые были внесены в file после добавления файла в отслеживаемое
"git mv name newname" - переименовать файл
"git diff" - проверить изменения во всей директории
"git branch" - вывод существующих веток
"git checkout branch" - переход на ветку
"git branch name" - создать ветку
"git branch -d  name" - удалить ветку
